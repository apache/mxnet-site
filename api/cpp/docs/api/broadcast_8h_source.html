88 broadcast(const expr::Exp<SrcExp, DType, etype> &src, Shape<dimdst> shape) {
89  TypeCheckPass<dimcast < dimdst && ExpInfo<SrcExp>::kDim == 1>
90  ::Error_Expression_Does_Not_Meet_Dimension_Req();
