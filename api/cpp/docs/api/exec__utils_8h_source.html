46 /*
47  * \brief setup default-storage tblobs from source NDArrays. If any source NDArray has non-default
48  * storage, it creates a temp NDArray with default storage and uses the temp tblob. The
49  * function also records the indices of non-default source NDArrays and the indices of
50  * their corresponding temporary NDArrays in the temp array.
51  * \param src list of source NDArray
52  * \param blobs list of tblobs to return
53  * \param temp_src list of source NDArrays which requires temporary default storage representation
54  * \param temp_dst list of temporary destination NDArrays for default storage representation
55  * \param idx_map mapping from indices in source NDArrays to indices in temp_dst. When not set,
56  indices are not recorded
57  * \return true if any source NDArray need to cast storage
58  */
59 inline bool SetupDefaultBlobsIn(const std::vector<NDArray>& src,
60  const std::vector<NDArray> *bufs,
61  std::vector<TBlob> *blobs,
62  std::vector<NDArray> *temp_src,
63  std::vector<NDArray> *temp_dst,
64  std::unordered_map<uint32_t, uint32_t> *idx_map) {
65  bool require_cast = false;
66  for (size_t i = 0; i < src.size(); i++) {
67  const auto& nd = src[i];
68  if (!DEFAULT_DATA(nd)) {
69  (*idx_map)[i] = temp_dst->size();
70  NDArray temp = bufs != nullptr ? bufs->at(i) : NDArray(nd.shape(), nd.ctx(),
71  true, nd.dtype());
72 #if MXNET_USE_ONEDNN == 1
73  CHECK(temp.IsDefaultData());
74 #endif
75  temp_src->emplace_back(nd);
76  temp_dst->emplace_back(temp);
77  blobs->emplace_back(temp.data());
78  require_cast = true;
79  } else {
80  blobs->push_back(nd.data());
81  }
82  }
83  return require_cast;
84 }

86 inline bool SetupDefaultBlobsOut(const std::vector<NDArray>& src,
87  const std::vector<NDArray> *bufs,
88  std::vector<OpReqType> *req,
89  std::vector<TBlob> *blobs,
90  std::vector<NDArray> *temp_src,
91  std::vector<NDArray> *temp_dst) {
92  bool require_cast = false;
93  for (size_t i = 0; i < src.size(); i++) {
94  const auto& nd = src[i];
95 
96 #if MXNET_USE_ONEDNN == 1
97  if (req->at(i) == kWriteInplace && nd.IsMKLDNNData())
98  // If it's write inplace and the output array doesn't use the default
99  // layout, we'll generate a temporary output array below, which means
100  // the input array and the output array are no longer the same array.
101  // we should change the request type.
102  req->at(i) = kWriteTo;
103  // We have to make sure it's default storage and default layout.
104 #endif
105  if (!DEFAULT_DATA(nd)) {
106 #if MXNET_USE_ONEDNN == 1
107  NDArray temp;
108  if (bufs != nullptr) {
109  temp = bufs->at(i);
110  } else if (kAddTo == req->at(i)) {
111  temp = nd.IsMKLDNNData()? nd.Reorder2Default() : nd;
112  } else {
113  temp = NDArray(nd.shape(), nd.ctx(), true, nd.dtype());
114  }
115  CHECK(temp.IsDefaultData());
116 #else
117  NDArray temp = bufs != nullptr ? bufs->at(i) : NDArray(nd.shape(), nd.ctx(),
118  true, nd.dtype());
119 #endif
120  temp_src->emplace_back(nd);
121  temp_dst->emplace_back(temp);
122  blobs->emplace_back(temp.data());
123  require_cast = true;
124  } else {
125  blobs->push_back(nd.data());
126  }
127  }
128  return require_cast;
129 }

139  const std::vector<NDArray> &ndoutputs,

141  const std::vector<NDArray> *out_bufs,
142  std::vector<OpReqType> *req,
143  std::vector<TBlob> *input_blobs,
144  std::vector<TBlob> *output_blobs,

162  post_temp_dst->push_back(ndinputs[idx]);

195 inline bool SameType(const nnvm::NodeAttrs& attrs,
196  std::vector<int> *iattr,
197  std::vector<int> *oattr) {
198  int def_v = -1;
199  for (int v : *oattr) {
200  if (v != -1) {
201  def_v = v; break;
202  }
203  }
204  if (def_v == -1) {
205  for (int v : *iattr) {
206  if (v != -1) {
207  def_v = v; break;
208  }
209  }
210  }
211  if (def_v == -1) return false;
212  for (int& v : *oattr) {
213  v = def_v;
214  }
215  for (int& v : *iattr) {
216  v = def_v;
217  }
218  return true;
219 }

227 inline bool DefaultStorageType(const nnvm::NodeAttrs& attrs,
228  const int dev_mask,
229  DispatchMode* dispatch_mode,
230  std::vector<int> *iattr,
231  std::vector<int> *oattr) {

234  if (v == -1) v = kDefaultStorage;
235  if (v != kDefaultStorage) fallback = true;
236  }

238  if (v == -1) v = kDefaultStorage;
239  if (v != kDefaultStorage) fallback = true;
240  }
241  if (*dispatch_mode == DispatchMode::kUndefined) {
242  if (fallback) {
243  *dispatch_mode = DispatchMode::kFComputeFallback;
244  } else {
245  *dispatch_mode = DispatchMode::kFCompute;
246  }
247  }
248  return true;

284 inline void LogMemoryPlan(const nnvm::Graph& g) {
285  const auto &idx = g.indexed_graph();
286  const auto& vshape = g.GetAttr<mxnet::ShapeVector>("shape");
287  const auto& vtype = g.GetAttr<nnvm::DTypeVector>("dtype");
288  // find node range
289  uint32_t node_start = 0, node_end = idx.num_nodes();
290  if (g.attrs.count("node_range")) {
291  const auto& range = g.GetAttr<std::pair<uint32_t, uint32_t> >("node_range");
292  node_start = range.first;
293  node_end = range.second;
294  }
295  for (uint32_t nid = node_start; nid < node_end; ++nid) {
296  const auto& inode = idx[nid];
297  if (inode.source->is_variable()) {
298  LOG(INFO) << "node " << nid << " var";
299  } else {
300  LOG(INFO) << "node " << nid << " " << inode.source->attrs.op->name;
301  for (const auto& e : inode.inputs) {
302  auto eid = idx.entry_id(e);
303  size_t kilo_bytes = vshape[eid].Size() * mshadow::mshadow_sizeof(vtype[eid]) / 1024;
304  LOG(INFO) << "\t\tinput " << eid << ": " << vshape[eid] << " ("
305  << kilo_bytes << " KB)";
306  }
307  for (uint32_t index = 0; index < inode.source->num_outputs(); ++index) {
308  uint32_t eid = idx.entry_id(nid, index);
309  size_t kilo_bytes = vshape[eid].Size() * mshadow::mshadow_sizeof(vtype[eid]) / 1024;
310  LOG(INFO) << "\t\toutput " << eid << ": " << vshape[eid] << " ("
311  << kilo_bytes << " KB)";
312  }
313  }
314  }
315 }

379 inline NDArray ReshapeOrCreate(const std::string& name,
380  const mxnet::TShape& dest_arg_shape,
381  const int dest_arg_dtype,
382  const NDArrayStorageType dest_arg_stype,
383  const Context& ctx,
384  std::unordered_map<std::string, NDArray>* shared_buffer,
385  bool enable_row_sparse_sharing) {
386  bool stype_shareable = dest_arg_stype == kDefaultStorage;
387  if (enable_row_sparse_sharing) {
388  stype_shareable = stype_shareable || dest_arg_stype == kRowSparseStorage;
389  }
390  auto it = shared_buffer->find(name);
391  if (it != shared_buffer->end()) {
392  // check if size is large enough for sharing
393  bool size_shareable = it->second.shape().Size() >= dest_arg_shape.Size();
394  if (size_shareable && stype_shareable) { // memory can be reused
395  CHECK_EQ(it->second.dtype(), dest_arg_dtype)
396  << "Requested arg array's dtype does not match that of the reusable ndarray";
397  CHECK_EQ(it->second.storage_type(), dest_arg_stype)
398  << "Requested arg array's stype does not match that of the reusable ndarray";
399  return it->second.Reshape(dest_arg_shape);
400  } else if (stype_shareable) {
401  LOG(WARNING) << "Bucketing: data " << name << " has a shape " << dest_arg_shape
402  << ", which is larger than already allocated shape " << it->second.shape()
403  << ". Need to re-allocate. Consider putting default bucket key to be "
404  << "the bucket taking the largest input for better memory sharing.";
405  // size is not large enough, creating a larger one for sharing
406  // the NDArrays in shared_buffer are guaranteed to be of shareable storages
407  it->second = InitZeros(dest_arg_stype, dest_arg_shape, ctx, dest_arg_dtype);
408  return it->second;
409  } else {
410  // not shareable storage
411  return InitZeros(dest_arg_stype, dest_arg_shape, ctx, dest_arg_dtype);
412  }
413  } else {
414  auto ret = InitZeros(dest_arg_stype, dest_arg_shape, ctx, dest_arg_dtype);
415  if (stype_shareable) {
416  shared_buffer->emplace(name, ret);
417  }
418  return ret;
419  } // if (it != shared_buffer->end())
420 }

426 inline nnvm::Graph AssignContext(nnvm::Graph g,
427  const Context& default_ctx,
428  const std::map<std::string, Context>& ctx_map,
429  const std::vector<Context>& in_arg_ctxes,
430  const std::vector<Context>& arg_grad_ctxes,
431  const std::vector<Context>& aux_state_ctxes,
432  const std::vector<OpReqType>& grad_req_types,
433  size_t num_forward_inputs,
434  size_t num_forward_outputs) {
435  const auto& idx = g.indexed_graph();
436  const auto& mutable_nodes = idx.mutable_input_nodes();
437  // default use default context.
438  if (ctx_map.size() == 0) {
439  g.attrs["context"] = std::make_shared<nnvm::any>(
440  exec::ContextVector(idx.num_nodes(), default_ctx));
441  for (const auto& x : in_arg_ctxes) {
442  CHECK(x == default_ctx)
443  << "Input array is in " << x << " while binding with ctx=" << default_ctx
444  << ". All arguments must be in global context (" << default_ctx
445  << ") unless group2ctx is specified for cross-device graph.";
446  }
447  for (const auto& x : arg_grad_ctxes) {
448  CHECK(x == default_ctx)
449  << "Gradient array is in " << x << " while binding with ctx="
450  << default_ctx << ". All gradients must be in global context (" << default_ctx
451  << ") unless group2ctx is specified for cross-device graph.";
452  }
453  return g;
454  }
455 
456  // otherwise, use context assignment.
457  std::map<Context, int> ctx2id; // map ctx to device id
458  std::vector<Context> ctx_list; // index is device id
459  nnvm::DeviceVector device(idx.num_nodes(), -1); // index is node id
460  nnvm::DeviceAssignMap device_map; // map arg name to device id
461 
462  // loop through the user input ctx_map and
463  // populate maps and lists
464  for (auto &kv : ctx_map) {
465  if (ctx2id.count(kv.second) == 0) { // if context has no device id, create one
466  ctx2id[kv.second] = static_cast<int>(ctx_list.size()); // assign device id to ctx
467  ctx_list.push_back(kv.second); // save ctx to the list
468  }
469  // assign device id to to the arg name with the corresponding ctx
470  device_map[kv.first] = ctx2id.at(kv.second);
471  }
472 
473  // loop through all the rest of input nodes not specified
474  // in the ctx_map and populate maps and lists
475  size_t arg_top = 0, aux_top = 0;
476  for (size_t i = 0; i < num_forward_inputs; ++i) {
477  const uint32_t nid = idx.input_nodes().at(i);
478  Context ctx;
479  if (mutable_nodes.count(nid)) { // aux node is mutable
480  CHECK_LT(aux_top, aux_state_ctxes.size());
481  ctx = aux_state_ctxes[aux_top];
482  ++aux_top;
483  } else { // regular input node is immutable
484  CHECK_LT(arg_top, in_arg_ctxes.size());
485  ctx = in_arg_ctxes[arg_top];
486  ++arg_top;
487  }
488  if (ctx2id.count(ctx) == 0) { // if the current ctx is not in the map of ctx and device id
489  ctx2id[ctx] = static_cast<int>(ctx_list.size()); // assign the current ctx with device id
490  ctx_list.push_back(ctx); // save the current ctx in the list
491  }
492  device[nid] = ctx2id.at(ctx); // assign device id to the current node
493  }
494 
495  // loop through backward input nodes and populate maps and lists
496  // the backward input nodes is the gradient of the loss wrt the output
497  size_t arg_grad_offset = 0;
498  // keep an offset into the arg_grad_ctxes vector,
499  // since g.outputs exclude arg_grad whose req == null
500  CHECK_GE(grad_req_types.size(), g.outputs.size() - num_forward_outputs)
501  << "insufficient number of grad_reqs";
502  for (size_t i = num_forward_outputs; i < g.outputs.size(); ++i, ++arg_grad_offset) {
503  while (grad_req_types[arg_grad_offset] == kNullOp) ++arg_grad_offset;
504  const uint32_t nid = idx.outputs()[i].node_id;
505  Context ctx = arg_grad_ctxes[arg_grad_offset];
506  if (ctx2id.count(ctx) == 0) {
507  ctx2id[ctx] = static_cast<int>(ctx_list.size());
508  ctx_list.push_back(ctx);
509  }
510  int devid = ctx2id.at(ctx);
511  if (device[nid] != -1) {
512  CHECK_EQ(device[nid], devid) << "device of same output not equal to each other";
513  } else {
514  device[nid] = devid;
515  }
516  }
517 
518  g.attrs["device"] = std::make_shared<dmlc::any>(std::move(device));
519  g = nnvm::pass::PlaceDevice(g, "__ctx_group__", device_map, "_CrossDeviceCopy");
520  const auto& assigned_devices = g.GetAttr<nnvm::DeviceVector>("device");
521 
522  exec::ContextVector vcontext;
523  for (auto context : assigned_devices) {
524  if (context == -1) {
525  vcontext.push_back(default_ctx);
526  } else {
527  vcontext.push_back(ctx_list[context]);
528  }
529  }
530 
531  // after device planning, we should check again
532  // if the assigned device of gradient node
533  // corresponds to storage of grads
534  auto &new_idx = g.indexed_graph();
535  arg_grad_offset = 0;
536  for (size_t i = num_forward_outputs; i < g.outputs.size(); ++i, ++arg_grad_offset) {
537  while (grad_req_types[arg_grad_offset] == kNullOp) ++arg_grad_offset;
538  const uint32_t nid = new_idx.outputs()[i].node_id;
539  Context ctx = arg_grad_ctxes[arg_grad_offset];
540  CHECK(ctx == vcontext[nid])
541  << "Trying to save gradient to " << ctx
542  << " while its source node \"" << new_idx[nid].source->attrs.name
543  << "\" computes it on " << vcontext[nid]
544  << ". Check your ctx in NDArray allocation.";
545  }
546 
547  g.attrs["context"] = std::make_shared<nnvm::any>(std::move(vcontext));
548  return g;
549 }
