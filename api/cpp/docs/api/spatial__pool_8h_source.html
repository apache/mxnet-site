60  : src_(src), ksize_y_(ksize_y), ksize_x_(ksize_x),

62  Shape<srcdim> sshape = ShapeCheck<srcdim, SrcExp>::Check(src_);
63  CHECK(sshape[srcdim - 1] >= ksize_x && sshape[srcdim - 2] >= ksize_y)
64  << "PoolingExp: kernel must be smaller than image";
65  this->src_height_ = sshape[srcdim - 2];
66  this->src_width_ = sshape[srcdim - 1];
67  this->shape_ = sshape;
68  this->shape_[srcdim - 2] = (src_height_ - ksize_y) / kstride_y + 1;
69  this->shape_[srcdim - 1] = (src_width_ - ksize_x) / kstride_x + 1;
70  }

73  index_t ksize_y, index_t ksize_x, index_t kstride_y, index_t kstride_x)

135 template<typename Reducer, typename SrcExp, typename DType, int srcdim>
136 struct Plan<PoolingExp< Reducer, SrcExp, DType, srcdim>, DType> {
137  public:
138  explicit Plan(const PoolingExp<Reducer, SrcExp, DType, srcdim> &e)
139  : src_(MakePlan(e.src_)),
140  ksize_y_(e.ksize_y_), ksize_x_(e.ksize_x_),
141  kstride_y_(e.kstride_y_), kstride_x_(e.kstride_x_),
142  src_height_(e.src_height_), src_width_(e.src_width_),
143  new_height_(e.shape_[srcdim - 2]) {}
144  MSHADOW_XINLINE DType Eval(index_t i, index_t j) const {
145  using namespace std;
146  const index_t py = i % new_height_;
147  const index_t y_start = py * kstride_y_;
148  const index_t y_end = min(y_start + ksize_y_, src_height_);
149  const index_t px = j;
150  const index_t x_start = px * kstride_x_;
151  const index_t x_end = min(x_start + ksize_x_, src_width_);
152  const index_t c = i / new_height_;
153 
154  DType res; Reducer::SetInitValue(res);
155  for (index_t y = y_start; y < y_end; ++y) {
156  for (index_t x = x_start; x < x_end; ++x) {
157  Reducer::Reduce(res, src_.Eval(c * src_height_ + y, x));
158  }
159  }
160  return res;
161  }
162 
163  private:
164  Plan<SrcExp, DType> src_;
165  const index_t ksize_y_, ksize_x_, kstride_y_, kstride_x_;
166  const index_t src_height_, src_width_;
167  const index_t new_height_;
168 };
