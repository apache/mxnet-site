75 struct csr_indptr_check {
76  template<typename DType, typename IType>
77  MSHADOW_XINLINE static void Map(int i, DType* out, const IType* indptr,
78  const nnvm::dim_t end, const nnvm::dim_t idx_size) {
79  if (indptr[i+1] < 0 || indptr[i+1] < indptr[i] ||
80  (i == 0 && indptr[i] != 0) ||
81  (i == end - 1 && indptr[end] != idx_size))
82  *out = kCSRIndPtrErr;
83  }
84 };

90 struct csr_idx_check {
91  template<typename DType, typename IType, typename RType>
92  MSHADOW_XINLINE static void Map(int i, DType* out, const IType* idx,
93  const RType* indptr, const nnvm::dim_t ncols) {
94  for (RType j = indptr[i]; j < indptr[i+1]; j++) {
95  if (idx[j] >= ncols || idx[j] < 0 ||
96  (j < indptr[i+1] - 1 && idx[j] >= idx[j+1])) {
97  *out = kCSRIdxErr;
98  break;
99  }
100  }
101  }
102 };

108 struct rsp_idx_check {
109  template<typename DType, typename IType>
110  MSHADOW_XINLINE static void Map(int i, DType* out, const IType* idx,
111  const nnvm::dim_t end, const nnvm::dim_t nrows) {
112  if ((i < end && idx[i+1] <= idx[i])
113  || idx[i] < 0 || idx[i] >= nrows)
114  *out = kRSPIdxErr;
115  }

130 template<typename xpu>
131 void CheckFormatCSRImpl(const RunContext &rctx, const NDArray &input,
132  const TBlob &err_cpu, const bool full_check) {
133  using namespace op::mxnet_op;
134  CHECK_EQ(input.storage_type(), kCSRStorage)
135  << "CheckFormatCSRImpl is for CSRNDArray";
136  const mxnet::TShape shape = input.shape();
137  const mxnet::TShape idx_shape = input.aux_shape(csr::kIdx);
138  const mxnet::TShape indptr_shape = input.aux_shape(csr::kIndPtr);
139  const mxnet::TShape storage_shape = input.storage_shape();
140  if ((shape.ndim() != 2) ||
141  (idx_shape.ndim() != 1 || indptr_shape.ndim() != 1 || storage_shape.ndim() != 1) ||
142  (indptr_shape[0] != shape[0] + 1) ||
143  (idx_shape[0] != storage_shape[0])) {
144  MSHADOW_TYPE_SWITCH(err_cpu.type_flag_, DType, {
145  DType* err = err_cpu.dptr<DType>();
146  *err = kCSRShapeErr;
147  });
148  return;
149  }
150  if (full_check) {
151  MSHADOW_TYPE_SWITCH(err_cpu.type_flag_, DType, {
152  MSHADOW_IDX_TYPE_SWITCH(input.aux_type(csr::kIndPtr), RType, {
153  MSHADOW_IDX_TYPE_SWITCH(input.aux_type(csr::kIdx), IType, {
154  mshadow::Stream<xpu> *s = rctx.get_stream<xpu>();
155  NDArray ret_xpu = NDArray(mshadow::Shape1(1),
156  rctx.get_ctx(), false, err_cpu.type_flag_);
157  TBlob val_xpu = ret_xpu.data();
158  Kernel<set_to_int<kNormalErr>, xpu>::Launch(s, val_xpu.Size(), val_xpu.dptr<DType>());
159  Kernel<csr_indptr_check, xpu>::Launch(s, indptr_shape[0] - 1, val_xpu.dptr<DType>(),
160  input.aux_data(csr::kIndPtr).dptr<RType>(),
161  indptr_shape[0] - 1, idx_shape[0]);
162  // no need to check indices if indices are empty
163  if (idx_shape[0] != 0) {
164  Kernel<csr_idx_check, xpu>::Launch(s, indptr_shape[0] - 1, val_xpu.dptr<DType>(),
165  input.aux_data(csr::kIdx).dptr<IType>(),
166  input.aux_data(csr::kIndPtr).dptr<RType>(), shape[1]);
167  }
168  mshadow::Copy(err_cpu.get<cpu, 1, DType>(),
169  val_xpu.get<xpu, 1, DType>(s), s);
170  });
171  });
172  });
173  }
174 }

184 template<typename xpu>
185 void CheckFormatRSPImpl(const RunContext &rctx, const NDArray &input,
186  const TBlob &err_cpu, const bool full_check) {
187  using namespace op::mxnet_op;
188  CHECK_EQ(input.storage_type(), kRowSparseStorage)
189  << "CheckFormatRSPImpl is for RSPNDArray";
190  const mxnet::TShape idx_shape = input.aux_shape(rowsparse::kIdx);
191  if (idx_shape[0] != input.storage_shape()[0]) {
192  MSHADOW_TYPE_SWITCH(err_cpu.type_flag_, DType, {
193  DType* err = err_cpu.dptr<DType>();
194  *err = kRSPShapeErr;
195  });
196  return;
197  }
198  if (idx_shape[0] == 0) {
199  return;
200  }
201  if (full_check) {
202  MSHADOW_TYPE_SWITCH(err_cpu.type_flag_, DType, {
203  MSHADOW_IDX_TYPE_SWITCH(input.aux_type(rowsparse::kIdx), IType, {
204  mshadow::Stream<xpu> *s = rctx.get_stream<xpu>();
205  NDArray ret_xpu = NDArray(mshadow::Shape1(1),
206  rctx.get_ctx(), false, err_cpu.type_flag_);
207  TBlob val_xpu = ret_xpu.data();
208  Kernel<set_to_int<kNormalErr>, xpu>::Launch(s, val_xpu.Size(), val_xpu.dptr<DType>());
209 
210  Kernel<rsp_idx_check, xpu>::Launch(s, idx_shape[0],
211  val_xpu.dptr<DType>(), input.aux_data(rowsparse::kIdx).dptr<IType>(),
212  idx_shape[0] - 1, input.shape()[0]);
213  mshadow::Copy(err_cpu.get<cpu, 1, DType>(),
214  val_xpu.get<xpu, 1, DType>(s), s);
215  });
216  });
217  }

426 inline std::string operator_stype_string(const nnvm::NodeAttrs& attrs,
427  const int dev_mask,
428  const std::vector<int>& in_attrs,
429  const std::vector<int>& out_attrs) {
430  std::ostringstream os;
431  os << "operator = " << attrs.op->name
432  << "\ninput storage types = [";
433  for (const int attr : in_attrs) {
434  os << stype_string(attr) << ", ";
435  }
436  os << "]\n"
437  << "output storage types = [";
438  for (const int attr : out_attrs) {
439  os << stype_string(attr) << ", ";
440  }
441  os << "]\n"
442  << "params = {";
443  for (auto kv : attrs.dict) {
444  os << "\"" << kv.first << "\" : " << kv.second << ", ";
445  }
446  os << "}\n"
447  << "context.dev_mask = " << dev_type_string(dev_mask);
448  return os.str();

452 inline std::string operator_string(const nnvm::NodeAttrs& attrs,
453  const OpContext& ctx,
454  const std::vector<NDArray>& inputs,
455  const std::vector<OpReqType>& req,
456  const std::vector<NDArray>& outputs) {
457  std::string result = "";
458  std::vector<int> in_stypes;
459  std::vector<int> out_stypes;
460  in_stypes.reserve(inputs.size());
461  out_stypes.reserve(outputs.size());
462  auto xform = [](const NDArray arr) -> int { return arr.storage_type(); };
463  std::transform(inputs.begin(), inputs.end(), std::back_inserter(in_stypes), xform);
464  std::transform(outputs.begin(), outputs.end(), std::back_inserter(out_stypes), xform);
465  result += operator_stype_string(attrs, ctx.run_ctx.ctx.dev_mask(), in_stypes, out_stypes);
466  return result;
467 }

539 template<typename RandomIt, typename Compare>
540 void ParallelSortHelper(RandomIt first, size_t len,
541  size_t grainsize, const Compare& comp) {
542  if (len < grainsize) {
543  std::sort(first, first+len, comp);
544  } else {
545  std::thread thr(ParallelSortHelper<RandomIt, Compare>, first, len/2, grainsize, comp);
546  ParallelSortHelper(first+len/2, len - len/2, grainsize, comp);
547  thr.join();
548  std::inplace_merge(first, first+len/2, first+len, comp);
549  }
550 }
