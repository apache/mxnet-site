42 template<typename SrcExp, typename DType, int dimsrc, int dimdst>
43 struct BroadcastWithAxisExp:
44  public MakeTensorExp<BroadcastWithAxisExp<SrcExp, DType, dimsrc, dimdst>,
45  SrcExp, dimdst, DType> {
47  const SrcExp &src_;
49  index_t dst_last_;
51  index_t trailing_;
53  index_t size_;
55  index_t last_;
57  BroadcastWithAxisExp(const SrcExp &src, const int axis, const index_t size)
58  : src_(src), size_(size) {
59  bool keepdim = (dimsrc == dimdst);
60  Shape<dimsrc> src_shape = ShapeCheck<dimsrc, SrcExp>::Check(src_);
61  this->trailing_ = 1;
62 
63  if (!keepdim) {
64  CHECK(dimsrc > axis && axis >= -1) << "broadcast axis (no keepdim) out of bound, " <<
65  "axis must be between -1 and" << dimsrc - 1 << ", given=" << axis << ".";
66  for (int i = 0; i <= axis; ++i) {
67  this->shape_[i] = src_shape[i];
68  }
69  this->shape_[axis + 1] = size_;
70  for (int i = axis + 1; i < dimsrc; ++i) {
71  this->trailing_ *= src_shape[i];
72  this->shape_[i + 1] = src_shape[i];
73  }
74  } else {
75  CHECK(dimdst > axis && axis >= 0) << "broadcast axis (keepdim) out of bound, " <<
76  "axis must be between 0 and" << dimdst - 1 << ", given=" << axis << ".";
77  CHECK_EQ(src_shape[axis], 1U) << "Size of the dimension of the broadcasting axis must be 1" <<
78  " when keepdim is on, src_shape[" << axis << "]=" << src_shape[axis] << ".";
79  for (int i = 0; i <= axis - 1; ++i) {
80  this->shape_[i] = src_shape[i];
81  }
82  this->shape_[axis] = size_;
83  for (int i = axis + 1; i < dimdst; ++i) {
84  this->trailing_ *= src_shape[i];
85  this->shape_[i] = src_shape[i];
86  }
87  }
88 
89  this->last_ = src_shape[dimsrc - 1];
90  this->dst_last_ = this->shape_[dimdst - 1];
91  }
92 }; // struct BroadcastWithAxisExp

130 template<typename SrcExp, typename DType, int dimsrc>
131 struct BroadcastWithMultiAxesExp :
132  public MakeTensorExp<BroadcastWithMultiAxesExp<SrcExp, DType, dimsrc>,
133  SrcExp, dimsrc, DType> {
135  const SrcExp &src_;
137  index_t dst_last_;
139  index_t axesnum_;
141  Shape<dimsrc> trailings_;
143  Shape<dimsrc> sizes_;
145  index_t last_;
147  template<typename TShape>
148  BroadcastWithMultiAxesExp(const SrcExp &src, const TShape& axes, const TShape& sizes)
149  : src_(src) {
150  Shape<dimsrc> src_shape = ShapeCheck<dimsrc, SrcExp>::Check(src_);
151  CHECK(axes.ndim() == sizes.ndim()) << "ndim of axes and sizes must be equal.";
152  this->axesnum_ = axes.ndim();
153  CHECK(this->axesnum_ <= dimsrc) << "Number of broadcasting axes must be smaller than"
154  "the source ndim, number of axes=" << this->axesnum_ << " dimsrc=" << dimsrc;
155  for (index_t i = 0; i < this->axesnum_; i++) {
156  CHECK(dimsrc > axes[i]) << "broadcast axis (keepdim) out of bound, " <<
157  "all axes must be between 0 and" << dimsrc - 1 << ", given axes[" << i << "] = " << axes[i]
158  << ".";
159  CHECK_EQ(src_shape[axes[i]], 1U) << "Size of the dimension of the broadcasting axis must be 1"
160  << ", src_shape[" << axes[i] << "]=" << src_shape[axes[i]] << ".";
161  if (i < this->axesnum_ - 1) {
162  CHECK(axes[i] < axes[i + 1]) << "The given axes must be in increasing order.";
163  }
164  }
165  for (index_t i = 0; i < dimsrc; i++) {
166  this->shape_[i] = src_shape[i];
167  this->sizes_[i] = 1;
168  this->trailings_[i] = 1;
169  }
170  for (index_t i = 0; i < this->axesnum_; i++) {
171  this->shape_[axes[i]] = sizes[i];
172  this->sizes_[i] = sizes[i];
173  }
174  for (index_t i = 0; i < this->axesnum_; i++) {
175  this->trailings_[i] = 1;
176  for (index_t j = axes[i] + 1; j < dimsrc; ++j) {
177  this->trailings_[i] *= this->shape_[j];
178  }
179  }
180  this->last_ = src_shape[dimsrc - 1];
181  this->dst_last_ = this->shape_[dimsrc - 1];
182  }
183 }; // struct BroadcastWithMultiAxesExp

212 template<typename SrcExp, typename DType, int etype, typename TShape>
213 inline BroadcastWithMultiAxesExp<SrcExp, DType, ExpInfo<SrcExp>::kDim>
214 broadcast_to(const Exp<SrcExp, DType, etype> &src, const TShape &target_shape) {
215  static const size_t dimsrc = ExpInfo<SrcExp>::kDim;
216  CHECK_EQ(target_shape.ndim(), dimsrc);
217  std::vector<index_t> axes_vec, sizes_vec;
218  Shape<dimsrc> src_shape = ShapeCheck<dimsrc, SrcExp>::Check(src.self());
219  for (size_t i = 0; i < dimsrc; ++i) {
220  if (src_shape[i] != target_shape[i]) {
221  CHECK_EQ(src_shape[i], 1U) << "broadcasting axis must have size 1, received shape="
222  << src_shape << " target_shape=" << target_shape;
223  axes_vec.push_back(i);
224  sizes_vec.push_back(target_shape[i]);
225  }
226  }
227  TShape axes = TShape(axes_vec.begin(), axes_vec.end());
228  TShape sizes = TShape(sizes_vec.begin(), sizes_vec.end());
229  return BroadcastWithMultiAxesExp<SrcExp, DType, ExpInfo<SrcExp>::kDim>(src.self(), axes, sizes);
230 }

235 template<typename SrcExp, typename DType, int dimsrc, int dimdst>
236 struct Plan<BroadcastWithAxisExp<SrcExp, DType, dimsrc, dimdst>, DType> {
237  public:
238  explicit Plan(const BroadcastWithAxisExp<SrcExp, DType, dimsrc, dimdst> &e)
239  : src_(MakePlan(e.src_)), dst_last_(e.dst_last_),
240  trailing_(e.trailing_), size_(e.size_), last_(e.last_) {}
241  MSHADOW_XINLINE DType Eval(index_t i, index_t j) const {
242  index_t x = (i * dst_last_ + j) / trailing_ / size_;
243  index_t y = (i * dst_last_ + j) % trailing_;
244  index_t z = x * trailing_ + y;
245  return src_.Eval(z / last_, z % last_);
246  }
247 
248  private:
249  Plan<SrcExp, DType> src_;
250  const index_t dst_last_, trailing_, size_, last_;
251 };

253 template<typename SrcExp, typename DType, int dimsrc>
254 struct Plan<BroadcastWithMultiAxesExp<SrcExp, DType, dimsrc>, DType> {
255  public:
256  explicit Plan(const BroadcastWithMultiAxesExp<SrcExp, DType, dimsrc> &e)
257  : src_(MakePlan(e.src_)), dst_last_(e.dst_last_), last_(e.last_), axesnum_(e.axesnum_),
258  trailings_(e.trailings_), sizes_(e.sizes_) {}
259  MSHADOW_XINLINE DType Eval(index_t i, index_t j) const {
260  index_t indx = i * dst_last_ + j;
261  for (index_t p = 0; p < dimsrc; ++p) {
262  if (p >= axesnum_) {
263  break;
264  }
265  indx = (indx / trailings_[p] / sizes_[p]) * trailings_[p] + (indx % trailings_[p]);
266  }
267  return src_.Eval(indx / last_, indx % last_);
268  }
269 
270  private:
271  Plan<SrcExp, DType> src_;
272  const index_t dst_last_, last_, axesnum_;
273  const Shape<dimsrc> trailings_, sizes_;
274 };
