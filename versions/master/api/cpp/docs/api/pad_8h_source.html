106  MSHADOW_XINLINE DType Eval(index_t i, index_t j) const {
107  const index_t x = j;
108  const index_t y = i % new_height_;
109  const index_t c = i / new_height_;
110  if (y < pad_y_ || x < pad_x_) return static_cast<DType>(0);
111  const index_t h = y - pad_y_;
112  const index_t w = x - pad_x_;
113  if (h < src_height_ && w < src_width_) {
114  return src_.Eval(c * src_height_ + h, w);
115  } else {
116  return static_cast<DType>(0);
117  }
118  }
