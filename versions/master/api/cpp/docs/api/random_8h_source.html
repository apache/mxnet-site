221  template<int dim, typename PType>
222  inline void SampleGeneralizedNegativeBinomial(Tensor<cpu, dim, DType> *dst,
223  PType mu, PType alpha) {
224  if (alpha == PType(0)) {
225  SamplePoisson(dst, mu); // limit of Poisson
226  } else {
227  PType r(PType(1) / alpha);
228  PType beta = mu * alpha;
229  std::gamma_distribution<> dist_gamma(r, beta);
230  typedef typename std::conditional<std::is_integral<DType>::value, DType, int>::type GType;
231  SampleDistribution(dst,
232  [&](){ std::poisson_distribution<GType> dist_poisson(dist_gamma(rnd_engine_));
233  return static_cast<DType>(dist_poisson(rnd_engine_));});
234  }

442 template<int dim>
443 inline void Random<gpu, DType>::SampleUniform(
444  Tensor<gpu, dim, DType> *dst, DType a, DType b) {
445  if (a == 0.0f && b == 1.0f) {
446  if (dst->CheckContiguous()) {
447  this->GenUniform(dst->dptr_, dst->shape_.Size());
448  } else {
449  *dst = this->uniform(dst->shape_);
450  }
451  } else {
452  *dst = this->uniform(dst->shape_) * (b - a) + a;
453  }
454 }

458  Tensor<gpu, dim, DType> *dst, DType mu, DType sigma) {
459  // We need to check whether the shape size is even since CuRand supports only normal distribution
460  // generation of even number of elements.
461  if (dst->CheckContiguous() && (dst->shape_.Size() % 2 == 0)) {
462  this->GenGaussian(dst->dptr_, dst->shape_.Size(), mu, sigma);
463  } else {
464  *dst = this->gaussian(dst->shape_, mu, sigma);
465  }
