28 // Manually maintained list of LAPACK interfaces that can be used
29 // within MXNET. Conventions:
30 // - Interfaces must be compliant with lapacke.h in terms of signature and
31 // naming conventions so wrapping a function "foo" which has the
32 // signature
33 // lapack_int LAPACKE_foo(int, char, lapack_int, float* , lapack_int)
34 // within lapacke.h should result in a wrapper with the following signature
35 // int MXNET_LAPACK_foo(int, char, int, float* , int)
36 // Note that function signatures in lapacke.h will always have as first
37 // argument the storage order (row/col-major). All wrappers have to support
38 // that argument. The underlying fortran functions will always assume a
39 // column-major layout. It is the responsibility of the wrapper function
40 // to handle the (usual) case that it is called with data in row-major
41 // format, either by doing appropriate transpositions explicitly or using
42 // transposition options of the underlying fortran function.
43 // - It is ok to assume that matrices are stored in contiguous memory
44 // (which removes the need to do special handling for lda/ldb parameters
45 // and enables us to save additional matrix transpositions around
46 // the fortran calls).
47 // - It is desired to add some basic checking in the C++-wrappers in order
48 // to catch simple mistakes when calling these wrappers.
49 // - Must support compilation without lapack-package but issue runtime error in this case.

102 inline void flip<cpu, float>(int m, int n,
103  float *b, int ldb, float *a, int lda) {
104  for (int i = 0; i < m; ++i)
105  for (int j = 0; j < n; ++j)
106  b[j * ldb + i] = a[i * lda + j];

134  inline int mxnet_lapack_sposv(int matrix_layout, char uplo, int n, int nrhs,
135  float *a, int lda, float *b, int ldb) {
136  int info;
137  if (matrix_layout == MXNET_LAPACK_ROW_MAJOR) {
138  // Transpose b to b_t of shape (nrhs, n)
139  float *b_t = new float[nrhs * n];
140  flip<cpu, float>(n, nrhs, b_t, n, b, ldb);
141  sposv_(&uplo, &n, &nrhs, a, &lda, b_t, &n, &info);
142  flip<cpu, float>(nrhs, n, b, ldb, b_t, n);
143  delete [] b_t;
144  return info;
145  }
146  sposv_(&uplo, &n, &nrhs, a, &lda, b, &ldb, &info);
147  return info;
148  }
