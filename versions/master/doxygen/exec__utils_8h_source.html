35 /*
36  * \brief setup default-storage tblobs from source NDArrays. If any source NDArray has non-default
37  * storage, it creates a temp NDArray with default storage and uses the temp tblob. The
38  * function also records the indices of non-default source NDArrays and the indices of
39  * their corresponding temporary NDArrays in the temp array.
40  * \param src list of source NDArray
41  * \param blobs list of tblobs to return
42  * \param temp_src list of source NDArrays which requires temporary default storage representation
43  * \param temp_dst list of temporary destination NDArrays for default storage representation
44  * \param idx_map mapping from indices in source NDArrays to indices in temp_dst. When not set,
45  indices are not recorded
46  * \return true if any source NDArray need to cast storage
47  */
48 inline bool SetupDefaultBlobsIn(const std::vector<NDArray>& src,
49  const std::vector<NDArray> *bufs,
50  std::vector<TBlob> *blobs,
51  std::vector<NDArray> *temp_src,
52  std::vector<NDArray> *temp_dst,
53  std::unordered_map<uint32_t, uint32_t> *idx_map) {
54  bool require_cast = false;
55  for (size_t i = 0; i < src.size(); i++) {
56  auto& nd = src[i];
57  bool is_default = nd.storage_type() == kDefaultStorage;
58 #if MXNET_USE_MKLDNN == 1
59  // We have to make sure it's default storage and default layout.
60  is_default = nd.IsDefaultData();
61 #endif
62  if (!is_default) {
63  (*idx_map)[i] = temp_dst->size();
64  NDArray temp = bufs != nullptr ? bufs->at(i) : NDArray(nd.shape(), nd.ctx(),
65  true, nd.dtype());
66 #if MXNET_USE_MKLDNN == 1
67  CHECK(temp.IsDefaultData());
68 #endif
69  temp_src->emplace_back(nd);
70  temp_dst->emplace_back(temp);
71  blobs->emplace_back(temp.data());
72  require_cast = true;
73  } else {
74  blobs->push_back(nd.data());
75  }
76  }
77  return require_cast;
78 }

80 inline bool SetupDefaultBlobsOut(const std::vector<NDArray>& src,
81  const std::vector<NDArray> *bufs,
82  std::vector<OpReqType> *req,
83  std::vector<TBlob> *blobs,
84  std::vector<NDArray> *temp_src,
85  std::vector<NDArray> *temp_dst) {
86  bool require_cast = false;
87  for (size_t i = 0; i < src.size(); i++) {
88  auto& nd = src[i];
89  bool is_default = nd.storage_type() == kDefaultStorage;
90 #if MXNET_USE_MKLDNN == 1
91  if (req->at(i) == kWriteInplace && nd.IsMKLDNNData())
92  // If it's write inplace and the output array doesn't use the default
93  // layout, we'll generate a temporary output array below, which means
94  // the input array and the output array are no longer the same array.
95  // we should change the request type.
96  req->at(i) = kWriteTo;
97  // We have to make sure it's default storage and default layout.
98  is_default = nd.IsDefaultData();
99 #endif
100  if (!is_default) {
101  NDArray temp = bufs != nullptr ? bufs->at(i) : NDArray(nd.shape(), nd.ctx(),
102  true, nd.dtype());
103 #if MXNET_USE_MKLDNN == 1
104  CHECK(temp.IsDefaultData());
105 #endif
106  temp_src->emplace_back(nd);
107  temp_dst->emplace_back(temp);
108  blobs->emplace_back(temp.data());
109  require_cast = true;
110  } else {
111  blobs->push_back(nd.data());
112  }
113  }
114  return require_cast;
115 }

117 /*
118  * \brief setup default-storage tblobs for input and output NDArrays.
119  * If any NDArray has non-default storage,
120  * it creates a temp NDArray with default storage and uses the temp tblob. The
121  * function also records the indices of non-default source NDArrays and the indices of
122  * their corresponding temporary NDArrays in the temp array.
123  */
124 inline void SetupDefaultBlobsInOut(const std::vector<NDArray> &ndinputs,
125  const std::vector<NDArray> &ndoutputs,
126  const std::vector<NDArray> *in_bufs,
127  const std::vector<NDArray> *out_bufs,
128  std::vector<OpReqType> *req,
129  std::vector<TBlob> *input_blobs,
130  std::vector<TBlob> *output_blobs,
131  std::vector<NDArray> *pre_temp_src,
132  std::vector<NDArray> *pre_temp_dst,
133  std::vector<NDArray> *post_temp_src,
134  std::vector<NDArray> *post_temp_dst,
135  std::unordered_map<uint32_t, uint32_t> *in_temp_idx_map,
136  const std::vector<uint32_t> &mutate_idx) {
137  // populate input blobs
138  SetupDefaultBlobsIn(ndinputs, in_bufs, input_blobs, pre_temp_src, pre_temp_dst,
139  in_temp_idx_map);
140  // populate output blobs
141  SetupDefaultBlobsOut(ndoutputs, out_bufs, req, output_blobs, post_temp_dst,
142  post_temp_src);
143  // add mutable inputs to post temp list
144  for (const auto idx : mutate_idx) {
145  auto map_iter = in_temp_idx_map->find(idx);
146  if (map_iter != in_temp_idx_map->end()) {
147  post_temp_src->push_back(pre_temp_dst->at(map_iter->second));
148  post_temp_dst->push_back(ndinputs[idx]);
149  }
150  }
151 }

153 /*
154  * \brief cast the NDArrays in `src` and store the result in NDArrays in `dst`.
155  * This is only used for storage fallback in executor.
156  * \param src list of source NDArray to cast
157  * \param dst list of destionation NDArray which hold the result of cast_storage operation
158  * \param ctx operator context for cast_storage operation
159  */
160 inline void CastNonDefaultStorage(const std::vector<NDArray>& src,
161  const std::vector<NDArray>& dst,
162  const OpContext& ctx,
163  const bool is_gpu) {
164  CHECK_EQ(dst.size(), src.size());
165  for (size_t i = 0; i < src.size(); i++) {
166  if (is_gpu) {
167 #if MXNET_USE_CUDA
168  CastStorageDispatch<gpu>(ctx, src[i], dst[i]);
169 #else
170  LOG(FATAL) << MXNET_GPU_NOT_ENABLED_ERROR;
171 #endif
172  } else {
173  CastStorageDispatch<cpu>(ctx, src[i], dst[i]);
174  }
175  }

181 inline bool SameType(const nnvm::NodeAttrs& attrs,
182  std::vector<int> *iattr,
183  std::vector<int> *oattr) {
184  int def_v = -1;
185  for (int v : *oattr) {
186  if (v != -1) {
187  def_v = v; break;
188  }
189  }
190  if (def_v == -1) {
191  for (int v : *iattr) {
192  if (v != -1) {
193  def_v = v; break;
194  }
195  }
196  }
197  if (def_v == -1) return false;
198  for (int& v : *oattr) {
199  v = def_v;
200  }
201  for (int& v : *iattr) {
202  v = def_v;
203  }
204  return true;
205 }

213 inline bool DefaultStorageType(const nnvm::NodeAttrs& attrs,
214  const int dev_mask,
215  DispatchMode* dispatch_mode,
216  std::vector<int> *iattr,
217  std::vector<int> *oattr) {
218  bool fallback = false;
219  for (int& v : *oattr) {
220  if (v == -1) v = kDefaultStorage;
221  if (v != kDefaultStorage) fallback = true;
222  }
223  for (int& v : *iattr) {
224  if (v == -1) v = kDefaultStorage;
225  if (v != kDefaultStorage) fallback = true;
226  }
227  if (*dispatch_mode == DispatchMode::kUndefined) {
228  if (fallback) {
229  *dispatch_mode = DispatchMode::kFComputeFallback;
230  } else {
231  *dispatch_mode = DispatchMode::kFCompute;
232  }
233  }
234  return true;
235 }

270 inline void LogMemoryPlan(const nnvm::Graph& g) {
271  const auto &idx = g.indexed_graph();
272  const auto& vshape = g.GetAttr<nnvm::ShapeVector>("shape");
273  const auto& vtype = g.GetAttr<nnvm::DTypeVector>("dtype");
274  const auto& vstorage = g.GetAttr<nnvm::StorageVector>("storage_id");
275  // find node range
276  uint32_t node_start = 0, node_end = idx.num_nodes();
277  if (g.attrs.count("node_range")) {
278  const auto& range = g.GetAttr<std::pair<uint32_t, uint32_t> >("node_range");
279  node_start = range.first;
280  node_end = range.second;
281  }
282  for (uint32_t nid = node_start; nid < node_end; ++nid) {
283  const auto& inode = idx[nid];
284  if (inode.source->is_variable()) {
285  LOG(INFO) << "node " << nid << " var";
286  } else {
287  LOG(INFO) << "node " << nid << " " << inode.source->attrs.op->name;
288  for (const auto& e : inode.inputs) {
289  auto eid = idx.entry_id(e);
290  size_t kilo_bytes = vshape[eid].Size() * mshadow::mshadow_sizeof(vtype[eid]) / 1024;
291  LOG(INFO) << "\t\tinput " << eid << ": " << vshape[eid] << " ("
292  << kilo_bytes << " KB) -> " << storage_str(vstorage[eid]);
293  }
294  for (uint32_t index = 0; index < inode.source->num_outputs(); ++index) {
295  uint32_t eid = idx.entry_id(nid, index);
296  size_t kilo_bytes = vshape[eid].Size() * mshadow::mshadow_sizeof(vtype[eid]) / 1024;
297  LOG(INFO) << "\t\toutput " << eid << ": " << vshape[eid] << " ("
298  << kilo_bytes << " KB) -> " << storage_str(vstorage[eid]);
299  }
300  }
301  }
302 }

333  uint32_t node_start = 0, node_end = idx.num_nodes();
334  if (g.attrs.count("node_range")) {

336  node_start = range.first;

340  const auto& inode = idx[nid];
341  if (inode.source->is_variable()) {
342  LOG(INFO) << "node " << nid << " var";
343  } else {
344  LOG(INFO) << "node " << nid << " " << inode.source->attrs.op->name
345  << ": " << dispatch_mode_string(dispatch_modes[nid]);
346  for (const auto& e : inode.inputs) {
