51 /*
52  * \brief setup default-storage tblobs from source NDArrays. If any source NDArray has non-default
53  * storage, it creates a temp NDArray with default storage and uses the temp tblob. The
54  * function also records the indices of non-default source NDArrays and the indices of
55  * their corresponding temporary NDArrays in the temp array.
56  * \param src list of source NDArray
57  * \param blobs list of tblobs to return
58  * \param temp_src list of source NDArrays which requires temporary default storage representation
59  * \param temp_dst list of temporary destination NDArrays for default storage representation
60  * \param idx_map mapping from indices in source NDArrays to indices in temp_dst. When not set,
61  indices are not recorded
62  * \return true if any source NDArray need to cast storage
63  */
64 inline bool SetupDefaultBlobs(const std::vector<NDArray>& src,
65  std::vector<TBlob> *blobs,
66  std::vector<NDArray> *temp_src,
67  std::vector<NDArray> *temp_dst,
68  std::unordered_map<uint32_t, uint32_t> *idx_map = nullptr) {
69  bool require_cast = false;
70  for (size_t i = 0; i < src.size(); i++) {
71  auto& nd = src[i];
72  if (nd.storage_type() != kDefaultStorage) {
73  if (idx_map != nullptr) {
74  (*idx_map)[i] = temp_dst->size();
75  }
76  NDArray temp(nd.shape(), nd.ctx(), false, nd.dtype());
77  temp_src->emplace_back(nd);
78  temp_dst->emplace_back(temp);
79  blobs->emplace_back(temp.data());
80  require_cast = true;
81  } else {
82  blobs->push_back(nd.data());
83  }
84  }
85  return require_cast;
86 }

88 /*
89  * \brief cast the NDArrays in `src` and store the result in NDArrays in `dst`.
90  * This is only used for storage fallback in executor.
91  * When storage_fallback is false, and `MXNET_EXEC_STORAGE_FALLBACK` == 0,
92  * storage fallback is disallowed.
93  * \param src list of source NDArray to cast
94  * \param dst list of destionation NDArray which hold the result of cast_storage operation
95  * \param ctx operator context for cast_storage operation
96  * \param storage_fallback whether storage_fallback is allowed. When set to false,
97  * its value depends on `MXNET_EXEC_STORAGE_FALLBACK`.
98  */
99 template <typename xpu>
100 inline void CastNonDefaultStorage(const std::vector<NDArray>& src,
101  const std::vector<NDArray>& dst,
102  const OpContext& ctx,
103  bool storage_fallback = false) {
104  CHECK_GE(dst.size(), src.size());
105  if (src.size() == 0) return;
106  if (storage_fallback == false) {
107  storage_fallback = dmlc::GetEnv("MXNET_EXEC_STORAGE_FALLBACK", true);
108  }
109  if (storage_fallback == false) {
110  LOG(FATAL) << "Storage type conversion detected during execution. "
111  << "You are probably executing an operator which "
112  << "doesn't support NDArray inputs with non-default storage.";
113  }
114  for (size_t i = 0; i < src.size(); i++) {
115  CastStorageDispatch<xpu>(ctx, src[i], dst[i]);
116  }
117 }

168 template<typename RandomIt, typename Compare>
169 void ParallelSortHelper(RandomIt first, size_t len,
170  size_t grainsize, const Compare& comp) {
171  if (len < grainsize) {
172  std::sort(first, first+len, comp);
173  } else {
174  std::thread thr(ParallelSortHelper<RandomIt, Compare>, first, len/2, grainsize, comp);
175  ParallelSortHelper(first+len/2, len - len/2, grainsize, comp);
176  thr.join();
177  std::inplace_merge(first, first+len/2, first+len, comp);
178  }
179 }

190 template<typename RandomIt, typename Compare>
191 void ParallelSort(RandomIt first, RandomIt last, size_t num_threads, Compare comp) {
192  const auto num = std::distance(first, last);
193  size_t grainsize = std::max(num / num_threads + 5, static_cast<size_t>(1024*16));
194  ParallelSortHelper(first, num, grainsize, comp);
