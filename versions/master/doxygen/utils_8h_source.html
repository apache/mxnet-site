144 template<typename RandomIt, typename Compare>
145 void ParallelSortHelper(RandomIt first, size_t len,
146  size_t grainsize, const Compare& comp) {
147  if (len < grainsize) {
148  std::sort(first, first+len, comp);
149  } else {
150  std::thread thr(ParallelSortHelper<RandomIt, Compare>, first, len/2, grainsize, comp);
151  ParallelSortHelper(first+len/2, len - len/2, grainsize, comp);
152  thr.join();
153  std::inplace_merge(first, first+len/2, first+len, comp);
154  }
155 }

166 template<typename RandomIt, typename Compare>

168  const auto num = std::distance(first, last);
169  size_t grainsize = std::max(num / num_threads + 5, static_cast<size_t>(1024*16));
